```python
def _map_task_for_domain(config: Config) -> None:
    domain_map = _TASK_TYPE_MAP.get(config.domain)
    if domain_map is None:
        return
    current_task = config.plugin_settings.get("task")
    if current_task is not None and current_task in domain_map:
        config.plugin_settings["task"] = domain_map[current_task]
```

From src/mlforge/finetuning/__init__.py (lines 27-29, 94-117):
```python
_VALID_METRICS: set[str] = {
    "perplexity", "rouge1", "rougeL", "rouge2", "loss",
}

def validate_config(self, config: Config) -> list[str]:
    errors = []
    if config.metric not in self._VALID_METRICS:
        errors.append(f"Unknown fine-tuning metric: {config.metric}. Valid: {sorted(self._VALID_METRICS)}")
    if not config.plugin_settings.get("model_name"):
        errors.append("plugin_settings missing 'model_name' -- required for fine-tuning")
    return errors
```
