```python
def diagnose(
    y_true: np.ndarray,
    y_pred: np.ndarray,
    dates: np.ndarray | pd.DatetimeIndex,
    top_n: int = 5,
    period: int = 4,
) -> dict:
    """Diagnose prediction errors for actionable feedback.

    Returns
    -------
    dict with keys:
        worst_periods: list[dict] — top_n periods with highest |error|,
            each dict has {date, y_true, y_pred, error, abs_error}
        bias: dict — {direction: "over"|"under"|"neutral", magnitude: float}
            direction = "over" if mean(y_pred - y_true) > 0
            magnitude = mean(y_pred - y_true)
        error_growth_correlation: float — Pearson correlation between
            |y_true| and |error|. Positive = errors grow with target size.
        seasonal_pattern: dict[str, float] — mean |error| grouped by
            quarter (Q1-Q4). Only present if dates are provided.
    """
```
